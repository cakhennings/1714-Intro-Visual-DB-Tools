Query 1: All Columns - Apartment & Building Tables, Sort on BuildingName:
SELECT Apartment.Apartment_ID, Apartment.Apt_Number, Apartment.Rent, Apartment.Beds, Apartment.Baths, Apartment.Building_ID, Apartment.Tenant_ID, Apartment.Manager_ID, Building.Building_Name, Building.Address_ID
FROM   Apartment INNER JOIN
             Building ON Apartment.Building_ID = Building.Building_ID
ORDER BY Building.Building_Name

Query 2: Person & Address Join - All Columns - Sort by LastName:
SELECT Address.Address_ID, Address.Street_1, Address.Street_2, Address.City, Address.State, Address.Zip, Person.First_Name, Person.Last_Name, Person.Middle_Initial, Person.Phone_Number
FROM   Address INNER JOIN
             Person ON Address.Address_ID = Person.Address_ID
ORDER BY Person.Last_Name

Query 3: Building & Address Join - Building Name and Address, Sort by City
SELECT Building.Building_Name, Address.Street_1, Address.Street_2, Address.City, Address.State, Address.Zip
FROM   Address INNER JOIN
             Building ON Address.Address_ID = Building.Address_ID
ORDER BY Address.City

Query 4: Receipt & Invoice Join - All columns
SELECT Invoice.InvoiceID, Invoice.InvoiceDate, Invoice.Apartment_ID, Invoice.DueDate, Receipt.Receipt_ID, Receipt.ReceiptDate, Receipt.ReceiptAmount
FROM   Invoice INNER JOIN
             Receipt ON Invoice.InvoiceID = Receipt.Invoice_ID

Query 5: LineItem & Product Join - city, name, price Filtern on InvoiceID=1
SELECT LineItem.Qty, Product.Name, Product.Price, LineItem.Invoice_ID
FROM   LineItem INNER JOIN
             Product ON LineItem.Product_ID = Product.Product_ID
WHERE (LineItem.Invoice_ID = 1)
